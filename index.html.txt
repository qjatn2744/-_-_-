                        return `
                            <div class="bg-white p-4 rounded-xl shadow-sm border border-gray-100 mb-3">
                                <div class="flex justify-between items-start mb-2">
                                    <h3 class="font-bold text-sm text-gray-800">${job ? job.school : '-'}</h3>
                                    ${badge}
                                </div>
                                <p class="text-[11px] text-gray-500 mb-1">📅 ${job ? job.date : '-'} | ⏰ ${job ? job.time : '-'}</p>
                                <div class="bg-indigo-50 text-indigo-700 text-xs font-bold p-2 rounded mt-2">
                                    ${a.subjectIndex + 1}번 슬롯: ${a.subjectName}
                                </div>
                            </div>`;
                    }).join('');
                    mainContent = `<div class="p-4 pb-20">${schedHtml}</div>`;
                } else if(currentView === 'inst-prof') {
                    headerTitle = "내 프로필";
                    let mySubjectNames = currentUser.subjects.map(s => s.name);
                    
                    let mySubjectsHtml = currentUser.subjects.length === 0 ? `<p class="text-xs text-gray-500 py-2">신청한 과목이 없습니다.</p>` : currentUser.subjects.map(sub => {
                        let badge = sub.status === '승인' ? `<span class="text-green-600 font-bold text-[10px]">승인완료</span>` 
                                  : (sub.status === '반려' ? `<span class="text-red-600 font-bold text-[10px]">반려됨</span>`
                                  : `<span class="text-yellow-600 font-bold text-[10px]">승인대기</span>`);
                        return `<div class="flex justify-between items-center border-b border-gray-100 py-2.5 last:border-0"><span class="text-xs font-medium text-gray-800 truncate pr-2">${sub.name}</span>${badge}</div>`;
                    }).join('');

                    let unappliedSubjects = allSubjects.filter(sub => !mySubjectNames.includes(sub));
                    let unappliedHtml = unappliedSubjects.length === 0 ? `<p class="text-xs text-gray-500 py-2">모든 강의를 신청했습니다.</p>` : unappliedSubjects.map(sub => `
                        <div class="flex justify-between items-center border-b border-gray-100 py-2 last:border-0">
                            <span class="text-[11px] font-medium text-gray-500 truncate pr-2 w-2/3">${sub}</span>
                            <button onclick="requestNewSubject('${sub}')" class="bg-white border border-indigo-300 text-indigo-600 font-bold py-1 px-2 rounded text-[10px] active:bg-indigo-50">+ 자격신청</button>
                        </div>`).join('');

                    mainContent = `
                        <div class="p-4 pb-20">
                            <div class="bg-white rounded-xl shadow-sm border border-gray-100 p-5 mb-4 flex items-center space-x-4">
                                <div class="w-16 h-16 bg-indigo-100 rounded-full flex items-center justify-center text-indigo-500 text-2xl font-bold shadow-inner">
                                    ${currentUser.name.charAt(0)}
                                </div>
                                <div>
                                    <h3 class="text-lg font-bold text-gray-800">${currentUser.name} <span class="text-xs font-normal text-gray-500">강사</span></h3>
                                    <p class="text-xs text-gray-500 mt-1">ID: ${currentUser.id}</p>
                                    <p class="text-xs text-gray-500">TEL: ${currentUser.phone || '미등록'}</p>
                                </div>
                            </div>
                            <div class="bg-white rounded-xl shadow-sm border border-gray-100 p-5 mb-4">
                                <h3 class="text-sm font-bold text-indigo-800 mb-3 border-b pb-2">나의 강의 자격 현황</h3>
                                ${mySubjectsHtml}
                            </div>
                            <div class="bg-gray-50 rounded-xl shadow-sm border border-gray-200 p-5 mb-4">
                                <h3 class="text-sm font-bold text-gray-700 mb-3 border-b border-gray-200 pb-2">미신청 강의 (추가 자격 신청)</h3>
                                ${unappliedHtml}
                            </div>
                        </div>`;
                }
            }

            // 하단 네비게이션 렌더링
            let navBar = "";
            if(currentUser.role === 'admin') {
                navBar = `
                    <nav class="bg-white border-t border-gray-200 flex justify-around pb-6 pt-2 absolute bottom-0 w-full z-50">
                        <button onclick="changeView('admin-jobs')" class="flex flex-col items-center ${currentView.includes('job') ? 'text-indigo-600' : 'text-gray-400'} p-2 transition w-1/3">
                            <span class="text-xl mb-1">📅</span><span class="text-[10px] font-bold">공고 관리</span>
                        </button>
                        <button onclick="changeView('admin-apps')" class="flex flex-col items-center ${currentView === 'admin-apps' ? 'text-indigo-600' : 'text-gray-400'} p-2 transition w-1/3">
                            <span class="text-xl mb-1">✅</span><span class="text-[10px] font-bold">지원 현황</span>
                        </button>
                        <button onclick="changeView('admin-inst')" class="flex flex-col items-center ${currentView === 'admin-inst' ? 'text-indigo-600' : 'text-gray-400'} p-2 transition w-1/3">
                            <span class="text-xl mb-1">👥</span><span class="text-[10px] font-bold">강사 관리</span>
                        </button>
                    </nav>`;
            } else {
                navBar = `
                    <nav class="bg-white border-t border-gray-200 flex justify-around pb-6 pt-2 absolute bottom-0 w-full z-50">
                        <button onclick="changeView('inst-jobs')" class="flex flex-col items-center ${currentView === 'inst-jobs' ? 'text-indigo-600' : 'text-gray-400'} p-2 transition w-1/3">
                            <span class="text-xl mb-1">📅</span><span class="text-[10px] font-bold">출강 공고</span>
                        </button>
                        <button onclick="changeView('inst-sched')" class="flex flex-col items-center ${currentView === 'inst-sched' ? 'text-indigo-600' : 'text-gray-400'} p-2 transition w-1/3">
                            <span class="text-xl mb-1">📋</span><span class="text-[10px] font-bold">내 일정</span>
                        </button>
                        <button onclick="changeView('inst-prof')" class="flex flex-col items-center ${currentView === 'inst-prof' ? 'text-indigo-600' : 'text-gray-400'} p-2 transition w-1/3">
                            <span class="text-xl mb-1">👤</span><span class="text-[10px] font-bold">프로필</span>
                        </button>
                    </nav>`;
            }

            root.innerHTML = `
                <header class="bg-indigo-600 text-white p-4 shadow-md flex justify-between items-center z-50 shrink-0">
                    <h1 class="font-bold text-lg tracking-tight">${headerTitle}</h1>
                    <button onclick="handleLogout()" class="text-xs bg-indigo-500 hover:bg-indigo-400 px-3 py-1.5 rounded-full font-bold transition">로그아웃</button>
                </header>
                <main class="flex-1 overflow-y-auto bg-gray-100 relative">
                    ${mainContent}
                </main>
                ${navBar}
            `;
        }

        // --- 3. 동작 로직 ---
        function changeView(v) { currentView = v; renderApp(); }
        function goHome() { if(!currentUser) changeView('login'); else if(currentUser.role === 'admin') changeView('admin-jobs'); else changeView('inst-jobs'); }
        function handleLogin() { const id = document.getElementById('loginId').value; const pw = document.getElementById('loginPw').value; const user = users.find(u => u.id === id && u.pw === pw); if (user) { currentUser = user; goHome(); } else { alert('아이디/비밀번호 오류입니다.'); } }
        function handleLogout() { currentUser = null; changeView('login'); }

        // 회원가입 관련
        function sendVerificationCode() {
            const email = document.getElementById('regEmail').value;
            if(!email.includes('@')) return alert("올바른 이메일 형식이 아닙니다.");
            tempRegData.sentCode = Math.floor(100000 + Math.random() * 900000).toString();
            alert(`[달꿈] 인증번호 발송 완료.\n테스트 번호: ${tempRegData.sentCode}`);
            document.getElementById('verificationInputArea').classList.remove('hidden');
        }
        function verifyEmailCode() {
            if(document.getElementById('regEmailCode').value === tempRegData.sentCode) {
                tempRegData.emailVerified = true;
                let msg = document.getElementById('verificationMsg'); msg.innerText = "✅ 인증되었습니다."; msg.className = "text-xs mt-1 text-green-600 font-bold";
            } else alert("인증번호 불일치.");
        }
        function goToRegisterStep2() {
            const name = document.getElementById('regName').value; const pw = document.getElementById('regPw').value; const phone = document.getElementById('regPhone').value;
            if(!name || !pw || !phone) return alert("필수 입력 누락");
            if(!tempRegData.emailVerified) return alert("이메일 인증 필수");
            const digitsOnly = phone.replace(/[^0-9]/g, '');
            tempRegData.name = name; tempRegData.pw = pw; tempRegData.phone = phone; tempRegData.last4 = digitsOnly.slice(-4);
            changeView('register2');
        }
        function submitRegistration() {
            let selectedSubjects = [];
            allSubjects.forEach((sub, index) => { if(document.getElementById(`chk_sub_${index}`).checked) selectedSubjects.push({ name: sub, status: '대기', reason: '가입 시 선택' }); });
            if(selectedSubjects.length === 0) return alert("최소 1개 수업 선택 필수");
            const finalId = tempRegData.name + tempRegData.last4;
            users.push({ id: finalId, pw: tempRegData.pw, name: tempRegData.name, phone: tempRegData.phone, role: 'instructor', subjects: selectedSubjects });
            alert(`가입 신청 완료!\n아이디: ${finalId}`); changeView('login');
        }

        // 공고 등록 및 관리 (Admin)
        window.addSubjectField = function() {
            const container = document.getElementById('jobSubjectsContainer'); const div = document.createElement('div'); div.className = "flex space-x-2";
            let options = allSubjects.map(sub => `<option value="${sub}">${sub}</option>`).join('');
            div.innerHTML = `<select class="flex-1 border border-gray-300 p-2 text-sm rounded-lg outline-none focus:border-indigo-500 job-subject-select">${options}</select><button onclick="this.parentElement.remove()" class="text-red-500 bg-red-50 px-3 rounded-lg font-bold">X</button>`;
            container.appendChild(div);
        }
        function submitNewJob() {
            const date = document.getElementById('jobDate').value, school = document.getElementById('jobSchool').value, target = document.getElementById('jobTarget').value, fee = document.getElementById('jobFee').value, time = document.getElementById('jobTime').value;
            const subjectSelects = document.querySelectorAll('.job-subject-select'); const subjectArray = Array.from(subjectSelects).map(select => select.value);
            if(!date || !school || subjectArray.length === 0) return alert("일정, 학교, 모집수업은 필수");
            jobs.push({ id: Date.now(), date, school, target, fee, time, subjects: subjectArray, isOpen: false });
            alert('공고 등록(대기 상태) 완료'); changeView('admin-jobs');
        }
        function forceOpenJob(id) { let j = jobs.find(x => x.id === id); if(j) { j.isOpen = true; renderApp(); } }
        function manageApplication(id, st) { let a = applications.find(x => x.appId === id); if(a) { a.status = st; renderApp(); } }
        function updateInstSubStatus(iId, sN, st) { let ins = users.find(u => u.id === iId); if(ins) { let s = ins.subjects.find(x => x.name === sN); if(s) { s.status = st; renderApp(); } } }

        // 강사 지원 및 신청 (Instructor)
        function applyForSubject(jId, sN, sIdx) { applications.push({ appId: Date.now(), jobId: jId, instructorId: currentUser.id, subjectName: sN, subjectIndex: sIdx, status: '대기' }); alert('지원 완료!'); renderApp(); }
        function requestNewSubject(sN) { let r = prompt(`[${sN}] 신청 사유:`); if(r && r.trim()!=='') { currentUser.subjects.push({ name: sN, status: '대기', reason: r }); alert('신청 완료!'); renderApp(); } }
